  // Refrescar cards/tabla solo cuando un batch TERMINA (no por cada archivo individual)
  const prevSessionsRef = useRef<typeof sessions>([])
  useEffect(() => {
    const prev = prevSessionsRef.current
    const curr = sessions
    // Detectar si alguna sesión acaba de terminar (procesando → no procesando)
    const alguienTermino = prev.some(ps => {
      const cs = curr.find(c => c.id === ps.id)
      return ps.procesando && cs && !cs.procesando
    })
    // O si hay nuevos archivos procesados (para actualizar al final)
    const totalCompletedNow = curr.reduce((acc, s) =>
      acc + (s.procesando ? 0 : s.log.length), 0)
    const totalCompletedPrev = prev.reduce((acc, s) =>
      acc + (s.procesando ? 0 : s.log.length), 0)

    if (alguienTermino || totalCompletedNow > totalCompletedPrev) {
      setRefreshTick(x => x + 1)
    }
    prevSessionsRef.current = curr
  }, [sessions])
